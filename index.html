variables

primitive types

  • string

    let firstName = "Ricardo";
    console.log(typeof firstName)

  • number

    let age = 25;
    console.log(typeof age)

  • boolean

    let use_credi= false;
    console.log (typeof use_credi)

  • undefined

    let phone;
    console.log(typeof phone)

reference types: objects

    let list = ["mick", "sebastian", "max", "lando","carlos"];
